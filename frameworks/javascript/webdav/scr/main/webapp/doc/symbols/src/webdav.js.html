 98 /**
 99  * opens XMLHttp request and sets its header 'Authorization' in case of this.username is not empty
100  * @private
101  * @param {String} name - the name of webdav method, e.g. "PROPFIND"
102  * @param {Function} client - client side handler-method of asynchronous process 
103  * @param {String} path - path, e.g. '/foo/example.txt' (on default = '')
104  * @param {Object} path - path, e.g. '/foo/example.txt' (on default = '')
105  * @return {Object} request
106  */
107 Webdav.prototype.openRequest = function(handler, method, path, additionalHeaders) {
108     var request = this.getRequest();
109 	if (request === null) {
110 		return null;
111 	}
112 
113    	/* prepare a request */
114     request.onreadystatechange = this.wrapperHandler(handler, request);
115 	var url = encodeURI(this.url + path);
116 
117 	request.open(method, url, this.asynchronous);
118 
119     // refuse all encoding, since the browsers don't seem to support it...
120     request.setRequestHeader('Accept-Encoding', ' ');
121 
122 	// set header for authorization
123 	if (this.username) {
124 		var auth = make_base_auth(this.username, this.password);
125 		request.setRequestHeader('Authorization', auth);
126 	}
127 	
128 	// set additional request headers
129 	if ( additionalHeaders ) {
130 		this.setAdditionalHeaders(request, additionalHeaders);
131 	}
132 
133 	return request;
134 }

136 /**
137  * wraps the handler with a callback
138  * @private 
139  * @param {Object} handler - functions to call when the request either succeeds or fails, and completes
140  * @param {Function} handler.onSuccess - will call if the request succeeds
141  * @param {Function} handler.onError - will call if the request fails
142  * @param {Function} handler.onComplete - will call if the request completes
143  * @param {Object} request - XMLHttp request object
144  * @return {Object} Handler object
145  */
146 Webdav.prototype.wrapperHandler = function(handler, request) {
147     var self = this;
148     function Handler() {
149 		handler = {
150 			onSuccess: handler.onSuccess || function(){},
151 			onError: handler.onError || function(){},
152 			onComplete: handler.onComplete || function(){}
153 		}
154         this.execute = function() {
155             if (request.readyState == 4) {
156 				var result = self.parseResponse(request);
157 				request = null;
158 				
159 				// Check to see if the request was successful
160 				if ( self.isSuccess(result.status) ) {
161 					handler.onSuccess.call('', result);
162 				} else {
163 					handler.onError.call('', result);
164 				}
165                 
166 				// Call the completion callback
167 				handler.onComplete.call('', result);
168             };
169         };
170     };
171     return (new Handler().execute);
172 };

174 /**
175  * parses the request and returns result-object
176  * @private 
177  * @param {Object} request - XMLHttpRequest object
178  * @return {Object} result :
179  * <br />- {Number} <b>result.status</b> = status of XMLHttp response 
180  * <br />- {Number} <b>result.statusstring</b> -  an explanation of status 
181  * <br />- {Object} <b>result.headers</b> - hash of request.getAllResponseHeaders() (e.g. header "Content-Type: test/plain" => result.headers['Content-Type'] = 'test/plain')
182  * <br />- {Object} <b>result.content</b> = XMLHttp response.responseXML if header['Content-Type'] consists 'xml'
183  * <br />- {String} <b>result.content</b> = XMLHttp response.responseText otherwise
184  */
185 Webdav.prototype.parseResponse = function(request){
186 	// define source of response content
187 	if ( request.getResponseHeader("content-type") &&
188 			request.getResponseHeader("content-type").indexOf("xml") >= 0 )
189 	{
190 		var content = request.responseXML;					
191 	} else {
192 		var content = request.responseText;					
193 	}
194 	
195 	var result = {
196 		status: request.status,
197 		statusstring: this.STATUS_CODES[request.status.toString()],
198 		headers: this.parseHeaders(request.getAllResponseHeaders()),
199 		content: content
200 	};
201 	
202 	return result; 
203 }

205 /**
206  * parses of XMLHttpRequest.getAllResponseHeaders()
207  * @private
208  * @param {String} headers - result of XMLHttpRequest.getAllResponseHeaders()
209  * @return {Object} hash of Headers (e.g. header "Content-Type: test/plain" => headers['Content-Type'] = 'test/plain')
210  */
211 Webdav.prototype.parseHeaders = function(headers) {
212 	var headerPattern = /([^:]+):.(.+)/;
213 	var lines = headers.split('\n');
214     var hash = {}; 
215 	var name = ''; 
216 	var value = '';
217 
218     for (var i = 0; i < lines.length; i++) {
219         var header = lines[i].match(headerPattern);
220 		if ( header != null && 
221 				typeof(header[1]) != 'undefined') 
222 		{
223 			name = header[1];
224 			if (typeof(header[2]) != 'undefined') {
225 				value = header[2];
226 			} else {
227 				value = '';
228 			}
229     	    hash[name] = value;
230 		}
231     };
232     
233     return hash;
234 };

315 /**
316  * returns string "Basic " + Base64 encoded pair "user: password"
317  * @private
318  * @param {String} user
319  * @param {String} password
320  */
321 function make_base_auth(user, password) {
322   var tok = user + ':' + password;
323   var hash = Base64.encode(tok);
324   return "Basic " + hash;
325 }

327 /**
328 *  Base64 encode / decode
329 *  http://www.webtoolkit.info/
330 * @private
331 **/
332 
333 var Base64 = {
334 
335     // private property
336     _keyStr : "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=",
337 
338     // public method for encoding
339     encode : function (input) {
340         var output = "";
341         var chr1, chr2, chr3, enc1, enc2, enc3, enc4;
342         var i = 0;
343 
344         input = Base64._utf8_encode(input);
345 
346         while (i < input.length) {
347 
348             chr1 = input.charCodeAt(i++);
349             chr2 = input.charCodeAt(i++);
350             chr3 = input.charCodeAt(i++);
351 
352             enc1 = chr1 >> 2;
353             enc2 = ((chr1 & 3) << 4) | (chr2 >> 4);
354             enc3 = ((chr2 & 15) << 2) | (chr3 >> 6);
355             enc4 = chr3 & 63;
356 
357             if (isNaN(chr2)) {
358                 enc3 = enc4 = 64;
359             } else if (isNaN(chr3)) {
360                 enc4 = 64;
361             }
362 
363             output = output +
364             this._keyStr.charAt(enc1) + this._keyStr.charAt(enc2) +
365             this._keyStr.charAt(enc3) + this._keyStr.charAt(enc4);
366 
367         }
368 
369         return output;
370     },
371 
372     // public method for decoding
373     decode : function (input) {
374         var output = "";
375         var chr1, chr2, chr3;
376         var enc1, enc2, enc3, enc4;
377         var i = 0;
378 
379         input = input.replace(/[^A-Za-z0-9\+\/\=]/g, "");
380 
381         while (i < input.length) {
382 
383             enc1 = this._keyStr.indexOf(input.charAt(i++));
384             enc2 = this._keyStr.indexOf(input.charAt(i++));
385             enc3 = this._keyStr.indexOf(input.charAt(i++));
386             enc4 = this._keyStr.indexOf(input.charAt(i++));
387 
388             chr1 = (enc1 << 2) | (enc2 >> 4);
389             chr2 = ((enc2 & 15) << 4) | (enc3 >> 2);
390             chr3 = ((enc3 & 3) << 6) | enc4;
391 
392             output = output + String.fromCharCode(chr1);
393 
394             if (enc3 != 64) {
395                 output = output + String.fromCharCode(chr2);
396             }
397             if (enc4 != 64) {
398                 output = output + String.fromCharCode(chr3);
399             }
400 
401         }
402 
403         output = Base64._utf8_decode(output);
404 
405         return output;
406 
407     },
408 
409     // private method for UTF-8 encoding
410     _utf8_encode : function (string) {
411         string = string.replace(/\r\n/g,"\n");
412         var utftext = "";
413 
414         for (var n = 0; n < string.length; n++) {
415 
416             var c = string.charCodeAt(n);
417 
418             if (c < 128) {
419                 utftext += String.fromCharCode(c);
420             }
421             else if((c > 127) && (c < 2048)) {
422                 utftext += String.fromCharCode((c >> 6) | 192);
423                 utftext += String.fromCharCode((c & 63) | 128);
424             }
425             else {
426                 utftext += String.fromCharCode((c >> 12) | 224);
427                 utftext += String.fromCharCode(((c >> 6) & 63) | 128);
428                 utftext += String.fromCharCode((c & 63) | 128);
429             }
430 
431         }
432 
433         return utftext;
434     },
435 
436     // private method for UTF-8 decoding
437     _utf8_decode : function (utftext) {
438         var string = "";
439         var i = 0;
440         var c = c1 = c2 = 0;
441 
442         while ( i < utftext.length ) {
443 
444             c = utftext.charCodeAt(i);
445 
446             if (c < 128) {
447                 string += String.fromCharCode(c);
448                 i++;
449             }
450             else if((c > 191) && (c < 224)) {
451                 c2 = utftext.charCodeAt(i+1);
452                 string += String.fromCharCode(((c & 31) << 6) | (c2 & 63));
453                 i += 2;
454             }
455             else {
456                 c2 = utftext.charCodeAt(i+1);
457                 c3 = utftext.charCodeAt(i+2);
458                 string += String.fromCharCode(((c & 15) << 12) | ((c2 & 63) << 6) | (c3 & 63));
459                 i += 3;
460             }
461 
462         }
463 
464         return string;
465     }
466 }
